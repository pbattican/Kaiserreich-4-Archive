HUN_remove_joseph_august = {
	HUN = {
		if = {
			limit = { has_character = AUS_joseph_august_von_habsburg }
			AUS_joseph_august_von_habsburg = {
				set_character_name = AUS_joseph_august_von_habsburg
				remove_all_country_leader_roles = yes
				set_nationality = AUS
			}
		}
	}
}
