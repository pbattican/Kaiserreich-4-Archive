country_event = { # Democracy Is Not Yet Lost
	id = pacificstates.0
	immediate = { log = "[GetDateText]: [Root.GetName]: event pacificstates.0 FROM = [From.GetName]" }
	title = pacificstates.0.t

	desc = {
		text = pacificstates.0.d.roosevelt
		trigger = { USA = { has_country_flag = USA_olson_was_president } }
	}
	desc = {
		text = pacificstates.0.d.mcnary
		trigger = { USA = { has_country_flag = USA_landon_was_president } }
	}
	desc = {
		text = pacificstates.0.d.willkie
		trigger = { USA = { has_country_flag = USA_garner_was_president } }
	}

	picture = GFX_report_event_USA_congress

	trigger = {
		USA = {
			OR = {
				has_country_flag = USA_olson_was_president
				has_country_flag = USA_garner_was_president
				has_country_flag = USA_landon_was_president
			}
		}
	}

	fire_only_once = yes
	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			remove_all_ministers = yes
			country_event = { id = pacificstates.204 days = 5 }
			country_event = { id = pacificstates.154 days = 2 }

			set_temp_variable = { days_elapsed = global.num_days }
			subtract_from_temp_variable = { days_elapsed = 706640 }

			set_temp_variable = { days_to_next_primary = 1622 }
			subtract_from_temp_variable = { days_to_next_primary = days_elapsed }
			country_event = { id = pacificstates.2 days = days_to_next_primary random_days = 3 }

		}
	}

	option = {
		name = pacificstates.0.a
		if = {
			limit = { USA = { has_country_flag = USA_olson_was_president } }
			set_country_flag = PSA_restored_roosevelt
			create_country_leader = {
				name = "Quentin Roosevelt"
				desc = "POLITICS_QUENTIN_ROOSEVELT_DESC"
				picture = "gfx/leaders/USA/Portrait_USA_Quentin_Roosevelt.png"
				expire = "1965.1.1"
				ideology = social_liberal_subtype
				traits = {
					appeaser
				}
			}
			set_politics = {
				ruling_party = social_liberal
				last_election = "1936.11.16"
				election_frequency = 48
				elections_allowed = yes
			}
			hidden_effect = {
				add_ideas = {
					PSA_Hiram_Johnson_hog_sli
					PSA_Franck_Roberts_Havenner_for_sli
					PSA_Jesse_Holman_Jones_eco_sli
					PSA_Bestor_Robinson_sec_sli
				}
			}
		}
		else_if = {
			limit = { USA = { has_country_flag = USA_garner_was_president } }
			set_country_flag = PSA_restored_willkie
			create_country_leader = {
				name = "Wendell Willkie"
				desc = "POLITICS_WENDELL_WILLKIE_DESC"
				picture = "gfx/leaders/USA/Portrait_USA_Wendell_Willkie_KR.png"
				expire = "1965.1.1"
				ideology = social_conservative_subtype
				traits = {
					smooth_talking_charmer
				}
			}
			set_politics = {
				ruling_party = social_conservative
				last_election = "1936.11.16"
				election_frequency = 48
				elections_allowed = yes
			}
			hidden_effect = {
				add_ideas = {
					PSA_Robert_Taylor_Jones_hog_sco
					PSA_Sheridan_Downey_for_sco
					PSA_William_Mulholland_eco_sco
					PSA_Henry_Fountain_Ashurst_sec_sco
				}
			}
		}
		else = {
			set_country_flag = PSA_restored_mcnary
			create_country_leader = {
				name = "Charles McNary"
				desc = "POLITICS_CHARLES_MCNARY_DESC"
				picture = "gfx/leaders/USA/Portrait_USA_Charles_L_McNary.png"
				expire = "1965.1.1"
